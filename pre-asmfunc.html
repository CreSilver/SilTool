Assembly funkce pro factorial v C: int faktorial(int t_N);

 module.asm; module.s

  global factorial

faktorial:
  ; edi vstup
  ; func int

  mov ecx, edi ; kopíruje vstup
  mov r8d, 1
  cmp edi, 0
  cmovbe eax, r8d ; jestliže vstup je 0 pak vracející hodnota je 1
  jbe .konec

  mov eax, ecx ; kopírování hodnoty
  dec ecx ; hodnota-1
  ; v eax je n
.cyklus:
  cmp ecx, 0
  jbe .konec ; pokud je hodnota 0 pak konec
  imul eax, ecx ; n*n-1
  dec ecx ; hodnota-1
  jge .cyklus

.konec:
  ret
