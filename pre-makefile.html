C, Assembler

Předvytvořený makefile pro práci se dvěmi soubory a to module.s a main.c

 Makefile

CC = gcc
AS = nasm

CFLAGS = -Wall -Wextra -no-pie -g
ASFLAGS = -f elf64 -g

TARGET = program
OBJS = main.o module.o

$(TARGET): (OBJS)
    (CC) $(CFLAGS) -o $(TARGET) (OBJS)

main.o: main.c
    (CC) (CFLAGS) -c main.c -o main.o
module.o: module.s
    (AS) $(ASFLAGS) module.s -o module.o

clean:
    rm -f $(OBJS) $(TARGET)
